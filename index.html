    /* NAV */

    window.addEventListener('scroll', () => {

    /* CANVAS animated grid */

    (function () {

      const canvas = document.getElementById('hero-canvas');

      const ctx = canvas.getContext('2d');

      let W, H, dots = [];


      function resize() {

        W = canvas.width  = canvas.offsetWidth;

        H = canvas.height = canvas.offsetHeight;

        dots = [];

        const cols = Math.ceil(W / 95), rows = Math.ceil(H / 95);

        for (let r = 0; r <= rows; r++)

          for (let c = 0; c <= cols; c++)

            dots.push({

              ox: c * (W / cols), oy: r * (H / rows),

              phase: Math.random() * Math.PI * 2,

              speed: 0.3 + Math.random() * 0.4,

              amp:   4   + Math.random() * 8,

              x: 0, y: 0,

            });

      }


      const COLS = () => Math.round(W / 95) + 1;

      let t = 0;

      function draw() {

        t += 0.005;

        ctx.clearRect(0, 0, W, H);

        const G = 'rgba(201,168,76,', cols = COLS();


        dots.forEach(d => {

          d.x = d.ox + Math.sin(t * d.speed + d.phase) * d.amp;

          d.y = d.oy + Math.cos(t * d.speed + d.phase * 1.3) * d.amp;

        });


        ctx.lineWidth = 0.4;

        dots.forEach((d, i) => {

          const col = i % cols, row = Math.floor(i / cols);

          if (col < cols - 1) {

            const n = dots[i + 1];

            ctx.strokeStyle = G + Math.max(0, 0.09 - Math.hypot(n.x - d.x, n.y - d.y) / 2500) + ')';

            ctx.beginPath(); ctx.moveTo(d.x, d.y); ctx.lineTo(n.x, n.y); ctx.stroke();

          }

          if (row > 0) {

            const n = dots[i - cols];

            ctx.strokeStyle = G + Math.max(0, 0.09 - Math.hypot(n.x - d.x, n.y - d.y) / 2500) + ')';

            ctx.beginPath(); ctx.moveTo(d.x, d.y); ctx.lineTo(n.x, n.y); ctx.stroke();

          }

          ctx.fillStyle = G + '0.18)';

          ctx.beginPath(); ctx.arc(d.x, d.y, 1, 0, Math.PI * 2); ctx.fill();

        });

        requestAnimationFrame(draw);

      }

      window.addEventListener('resize', resize);

      resize(); draw();

    })();


    /* ── SCROLL ANIMATIONS ── */

    document.body.classList.add('js-anim');


    function revealEl(el) { el.classList.add('in-view'); }


    function revealAlreadyInView() {

      document.querySelectorAll('.reveal, .industry-card').forEach(el => {

        const r = el.getBoundingClientRect();

        if (r.top < window.innerHeight && r.bottom > 0) revealEl(el);

      });

    }


    const revealObserver = new IntersectionObserver((entries) => {

      entries.forEach(entry => {

        if (entry.isIntersecting) {

          revealEl(entry.target);

          revealObserver.unobserve(entry.target);

        }

      });

    }, { threshold: 0.05 });


    document.querySelectorAll('.reveal').forEach(el => revealObserver.observe(el));


    const cardObserver = new IntersectionObserver((entries) => {

      entries.forEach(entry => {

        if (entry.isIntersecting) {

          entry.target.querySelectorAll('.industry-card').forEach((c, i) => {

            setTimeout(() => revealEl(c), i * 100);

          });

          cardObserver.unobserve(entry.target);

        }

      });

    }, { threshold: 0.05 });


    const grid = document.querySelector('.industries-grid');

    if (grid) cardObserver.observe(grid);


    requestAnimationFrame(revealAlreadyInView);


    setTimeout(() => {

      document.querySelectorAll('.reveal, .industry-card').forEach(revealEl);

    }, 2500);


    /* PROBLEM QUOTE — word-by-word scrub */

    function updateQuote() {

      const q = document.getElementById('problem-quote');

      if (!q) return;

      const r = q.getBoundingClientRect();

      const wh = window.innerHeight;

      const progress = Math.max(0, Math.min(1, (wh - r.top) / (wh * 0.6)));

      q.querySelectorAll('.rw').forEach((w, i, arr) => {

        w.classList.toggle('lit', progress >= i / arr.length);

      });

    }

    window.addEventListener('scroll', updateQuote, { passive: true });

    updateQuote();
